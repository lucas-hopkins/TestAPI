
    [Route("api/[controller]")]
    [ApiController]
    public class EmployeeController : ControllerBase
    {
        private readonly ActorRepository _aR;

        public EmployeeController(ActorRepository aR)
        {
            _aR = aR;
        }


        [HttpGet]
        [Route("api/[controller]")]
        public async Task<ActionResult<List<Actor>>> Index()
        {
            return await _aR.GetAll();
        }

        [HttpGet]
        [Route("api/[controller]/{id}")]
        public async Task<ActionResult<Actor>> GetByID(int id)
        {
            return await _aR.GetByID(id);
        }
    }
}
